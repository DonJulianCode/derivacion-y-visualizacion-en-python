# Funcin personalizada para mostrar expresiones simblicas con formato LaTeX
def mostrar_latex(expr, nombre_funcion="f"):
    """
    Muestra en pantalla la representacin LaTeX de una expresin simblica.

    Parmetros:
    - expr: expresin simblica de SymPy (por ejemplo, una funcin o derivada)
    - nombre_funcion: nombre que queremos que aparezca como etiqueta (por defecto, 'f')

    Resultado:
    - Muestra la funcin en formato matemtico bonito usando LaTeX en un notebook de Colab
    """
    latex_expr = sp.latex(expr)
    display(Math(rf"{nombre_funcion}(x) = {latex_expr}"))


# Definimos la variable simblica x, que representa nuestro eje horizontal
x = sp.Symbol('x')

# Definimos la funcin f(x) = (x - 3)^2
f = (x - 3)**2

# Mostramos la funcin original de manera elegante
mostrar_latex(f, "f")

# Calculamos la derivada de f con respecto a x
f_prime = sp.diff(f, x)

# Mostramos la derivada simblicamente
mostrar_latex(f_prime, "f'")

# Buscamos los puntos donde la derivada se anula, es decir, los puntos crticos
solucion = sp.solve(f_prime, x)

# Mostramos el resultado en consola y tambin en formato matemtico
print("Punto crtico donde f'(x) = 0:", solucion)
display(Math(rf"x = {solucion[0]}"))

$\displaystyle f'(x) = 2 x - 6$

# === Funcin simblica y derivada ===
x = sp.Symbol('x')  # Definimos x como variable simblica

# Definimos la funcin f(x) = (x - 3)^2
f_expr = (x - 3)**2

# Creamos una versin numrica de la funcin para usar con NumPy
f = sp.lambdify(x, f_expr, "numpy")

# Derivamos simblicamente para encontrar el punto crtico
f_prime = sp.diff(f_expr, x)

# Obtenemos el punto crtico (donde la derivada es cero)
punto_critico_simbolico = float(sp.solve(f_prime, x)[0])

# Evaluamos la funcin original en ese punto crtico
valor_minimo_simbolico = float(f_expr.subs(x, punto_critico_simbolico))

# === Definicin de la funcin objetivo para optimizacin numrica ===
def funcion_objetivo(x):
    """
    Funcin que queremos minimizar numricamente.

    Parmetros:
    - x: valor (o arreglo) sobre el que se evala la funcin

    Retorna:
    - Valor de (x - 3)^2, que tiene un mnimo en x = 3
    """
    return (x - 3)**2

# === Minimizacin numrica con SciPy ===
# Usamos un valor inicial arbitrario, como x = 0
resultado = minimize(funcion_objetivo, x0=0)

# Extraemos los resultados de la optimizacin
x_min_numerico = resultado.x[0]      # Valor de x donde se alcanza el mnimo
f_min_numerico = resultado.fun       # Valor mnimo de la funcin

# === Mostramos los resultados simblicos y numricos de forma elegante ===
display(Math(r"\text{Resultado simblico:}"))
display(Math(rf"x = {punto_critico_simbolico},\quad f(x) = {valor_minimo_simbolico}"))

display(Math(r"\text{Resultado numrico con SciPy:}"))
display(Math(rf"x \approx {x_min_numerico:.6f},\quad f(x) \approx {f_min_numerico:.6f}"))

# Tambin mostramos los resultados en consola para revisin rpida
print("")
print("Comparacin de resultados")
print("Simblico:", punto_critico_simbolico, "| Numrico:", x_min_numerico)
